快速开始

# 克隆仓库
git clone https://github.com/LiHongwei-cn/lihongwei-cn.git
cd lihongwei-cn/mundo-agent

# 创建虚拟环境
python3 -m venv venv
source venv/bin/activate

# 安装依赖
pip install -r requirements.txt

# 运行蒙多
python3 mundo.py

Cursor 集成

// .cursor/mcp.json
{
  "mcpServers": {
    "mundo-agent": {
      "command": "python3",
      "args": ["mundo-agent/mcp_server.py"]
    }
  }
}
